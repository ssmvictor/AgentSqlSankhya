# requirements.txt
# Dependências do projeto IA_Agno com indexação do Schema Sankhya

# Este repositório segue a estrutura:
# src/          - Código-fonte principal (schema_indexer, schema_optimizer, constantes)
# config/       - Configurações, prompts e schema Sankhya
# agents/       - Integrações com agentes (Agno, etc.)
# scripts/      - Scripts utilitários (manage_cache)
# tests/        - Testes automatizados
# docs/         - Documentação detalhada e histórico

# Para executar módulos diretamente:
# python -m src.schema_indexer
# python -m src.schema_optimizer
# python -m tests.test_indexer
# python -m scripts.manage_cache <comando>

# Framework de agentes (instalar separadamente quando disponível)
# agno

# Variáveis de ambiente e configurações
python-dotenv  # Carrega .env e facilita parâmetros customizados

# Contador de tokens para LLMs (opcional mas recomendado)
tiktoken  # Mede tokens durante otimizações de contexto

# Dependências opcionais
# streamlit  # Interface web (opcional)
# openai  # Cliente para APIs OpenAI
# anthropic  # Integração com Claude
# requests  # Chamadas HTTP genéricas

# COMANDOS ÚTEIS:
# Instalar dependências: pip install -r requirements.txt
# Executar indexador: python -m src.schema_indexer
# Executar otimizador: python -m src.schema_optimizer
# Executar agente (requer agno): python -m agents.agno_integration
# Executar testes: python -m tests.test_indexer
# Limpar cache: python -m scripts.manage_cache clear
# Ver info do cache: python -m scripts.manage_cache info
